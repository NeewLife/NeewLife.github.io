- 문제7. n개의 랜덤배열을 만든 후, 오름차순으로 정렬하여라.

randomNum3 = [];
for (let i = 0; i < 5; i++){
    randomNum3[i] = Math.floor(Math.random() * 100)
}
console.log(randomNum3);

function asc(arr) {
    for (let n = 1; n < arr.length; n++){
        for (let i = 1;i <arr.length; i++){
            if(arr[i] < arr[i-1]){
                var change = arr[i-1];
                arr[i-1] = arr[i];
                arr[i] = change;
            }console.log(arr);

    }
}
